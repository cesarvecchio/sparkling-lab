
go

create table sindico(
	id_sindico int primary key identity(1,1),
	nome_sindico varchar(45),
	sobrenome_sindico varchar(50),
	cpf_sindico char(11) not null unique,
	email_sindico varchar(60) not null unique,
	senha_sindico varchar(45) not null unique
);
go

create table status_sensor_luz(
	id_status_sensor_luz int primary key identity(1,1),
	descricao_status_sensor_luz varchar(45) not null unique,
);
go

create table status_sensor_proximidade(
	id_status_sensor_proximidade int primary key identity(1,1),
	descricao_status_sensor_proximidade varchar(45) not null unique,
);
go

create table condominio(
	id_condominio int primary key identity(1,1),
	nome_condominio varchar(45) not null,
	fk_sindico int foreign key references sindico(id_sindico)
);
go

create table sensor_luz(
	id_sensor_luz int primary key identity(1,1),
	local_sensor_luz varchar(45) not null,
	fk_condominio int foreign key references condominio(id_condominio),
	fk_status_sensor_luz int foreign key references status_sensor_luz(id_status_sensor_luz)
);
go

create table sensor_proximidade(
	id_sensor_proximidade int primary key identity(1,1),
	local_sensor_proximidade varchar(45) not null,
	fk_condominio int foreign key references condominio(id_condominio),
	fk_status_sensor_proximidade int foreign key references status_sensor_proximidade(id_status_sensor_proximidade)
);
go

create table grafico_sensor_luz(
	id_grafico_sensor_luz int primary key identity(1,1),
	data_sensor_luz date,
	hora_sensor_luz time,
	lumens decimal,
	fk_sensor_luz int foreign key references sensor_luz(id_sensor_luz)
);
go

create table grafico_sensor_proximidade(
	id_grafico_sensor_proximidade int primary key identity(1,1),
	data_sensor_proximidade date,
	hora_sensor_proximidade time,
	fk_sensor_proximidade int foreign key references sensor_proximidade(id_sensor_proximidade)
);
go
